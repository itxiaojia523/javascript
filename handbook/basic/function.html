    函数 函数的声明方式

    1. 利用函数关键字自定义函数（命名函数）： function 函数名(){}
    2. 函数表达式（匿名函数）：var 变量名 = function(){}

    JS引擎运行JS分两步：预解析和代码执行 预解析: 会把var和function提升到当前作用域的最前面 变量预解析（变量提升） 只提升变量声明不提升赋值 函数预解析（函数提升）

    //求任意个数的最大值
function getMax(){
  // arguments是伪数组，存储所有的实参;只有函数有 arguments
  var max = arguments[0]
  for (var i = 0; i < arguments.length; i++){
    max = max > arguments[i]? max:arguments[i]
  }
  // return的意思：函数名() = return后的内容 将return后的数据赋值给这个函数
  return max
}
console.log(getMax(11,23,5))
console.log(getMax(11,23,5,155))

function reverse(arr){
  var newArr = []
  for(i = arr.length - 1; i >= 0; i--){
    newArr[newArr.length] = arr[i]
  }
  return newArr
}
console.log(reverse(arr))

//冒泡排序，将一系列数据按一定顺序排列（小到大或大到小）
    //从小到大排列
var arr1 = [1,4,5,2,3]
function sort(arr) {
  for (i = 0; i < arr.length - 1; i++) {
    for (j = 0; j < arr.length - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        var temp = arr[j]
        arr[j] = arr[j + 1]
        arr[j + 1] = temp
      }
    }
  }
  return arr
}
// return的意思：函数名() = return后的内容 将return后的数据赋值给这个函数
var result = sort(arr1)
console.log(result)

//判断一个字符串中出现次数最多的字符，并统计次数
//算法：1.charAt遍历 2.将字符存给一个对象 3.判断对象是否有该属性，没有为1，有则+1 4.遍历新对象，获取最大值
var str = 'abcddeeffggg'
var obj = {}
for (var i = 0; i < str.length; i++) {
    var chars = str.charAt(i)
    if (obj[chars]){
      obj[chars]++
    }else{
      obj[chars] = 1
    }
}
var max = 0
var ch = ''
for (var k in obj){
  if (obj[k] > max){
    max = obj[k]
    ch = k
  }
}
console.log('出现最多为'+ch+'次数为：'+max)
