    必包 产生条件，函数嵌套且内部函数引用了外部函数变量，执行外部函数

    什么是必包 是一个内部函数的对象，含有被引用变量

    作用：延长局部变量生命周期， 使得外部能操作内部的变量 将函数作为返回值 将函数作为实参 定义模块

    遍历+监听 for(var i = 0;i <btns.length; i++){}

    for(var i = 0, length = btns.length;i <length; i++){} 第二种写法性能更好，length只计算一次

    for(var i = 0, length = btns.length;i <length; i++){ var btn = btns[i] btn.onclick = function (){ alert('第'+(i+1)+'个') //一直是4 } } console.log(i) /3 1.遍历完了以后 才执行函数 2.所以要给btn加一个下标 var btn = btns[i] //将btn所对应的下标保存在btn上 btn.index = i btn.onclick = function (){ alert('第'+(this.index+1)+'个') }

    闭包 for(var i = 0, length = btns.length;i <length; i++){ (function (i){ var btn = btns[i] btn.onclick = function (){ alert('第'+(i+1)+'个') } })(i) }

    如何产生闭包 1.嵌套的函数 2.内部函数，引用了父函数的变量 3.执行外部函数 注意：不用调用内部函数，仅调用外部函数 就产生了

    function f1(){ var a = 2 var b = 3 function fn2() { console.log(a) } } f1() 没引用b 所以闭包里没有b

    什么是闭包 理解1：是嵌套的内部函数（大多数人） 理解2：是包含被引用变量的对象（在内部函数中） 少数人 通过chrome debug程序查看

    常见的闭包 1.将函数作为另外一个函数的返回值 function f1(){ var a = 2 function fn2() { a++ console.log(a) } return fn2 } var f = fn1() //如果没有闭包 fn1执行，a就没了，执行f，就报错了 //这说明在反复执行函数的过程中 闭包中的数据没有消失！ f()// a = 3 f()// a = 4 f = null //让闭包变成垃圾对象，死亡

    2.将函数作为实参传递给另外一个函数 function showDelay(msg, time){ setTimeout(function (){ alert(msg) },time) } showDelay('maomao',2000)

    闭包的作用 1.保存局部变量（函数执行完后，仍存在内存中，延长了局部变量的生命周期） 2.在外部操作局部变量

    闭包的应用 1.定义JS模块 myModule -JS文件 -将所有函数和功能封装在一个函数中 -只向外暴露一个或多个方法的对象 ;function myModule(){ var msg = 'maomao' function doSomething(){ console.log('doSomething()' + msg.toUpperCase()) } function doOtherthing() { console.log('doOtherthing' + msg.toLowerCase()) } //向外暴露对象，给外部使用的方法 return { doSomething: doSomething, doOtherthing: doOtherthing } } var module = myModule() module.doSomething()

    或（推荐，更直接） ;(function myModule(){ var msg = 'maomao' function doSomething(){ console.log('doSomething()' + msg.toUpperCase()) } function doOtherthing() { console.log('doOtherthing' + msg.toLowerCase()) } //向外暴露对象，给外部使用的方法 window.myModule2 = { doSomething: doSomething, doOtherthing: doOtherthing } })() myModule2.doSomething()

    最终版 ;(function myModule(window){ var msg = 'maomao' function doSomething(){ console.log('doSomething()' + msg.toUpperCase()) } function doOtherthing() { console.log('doOtherthing' + msg.toLowerCase()) } //向外暴露对象，给外部使用的方法 window.myModule2 = { doSomething: doSomething, doOtherthing: doOtherthing } })(window) 如果没定义参数，不能压缩为w，与代码压缩相关

    闭包的缺点 1.函数内局部变量没有释放，占用内存事件会变长 2.容易造成内存泄漏 解决 1.能不用就不用 2.及时释放 即f = null

    内存溢出与内存泄漏 内存溢出： 当一个程序运行需要的内存超过剩余的内存空间，就出现内存溢出的错误 导致程序出现错误 内存泄漏： 占用的内存没有及时释放 内存泄漏积累多了就容易导致内存溢出

    常见的内存泄漏 意外的全局变量 函数内部 a = 3（全局） 与 var a = 3 （局部） 没有及时清理的计时器或回调函数 clearInterval 闭包 f = null

    var name = 'The Window' var obj = { name: 'My Obj', getName: function () { return function () { return this.name } } } alert(obj.getName()())//相当于这个函数直接调用 function(){}(),this是window

    var name = 'The Window' var obj = { name: 'My Obj', getName: function () { var that = this return function () { return that.name } } } alert(obj.getName()()) //that 是getName的this，this就是obj ！常用，将一个函数的this用变量存起来，最后用

    var b = fun(0).fun(1).fun(2).fun(3) //undefined 0 1 2 var c = fun(0).fun(1) undefined 0 c.fun(2) //1 c.fun(3) //1
